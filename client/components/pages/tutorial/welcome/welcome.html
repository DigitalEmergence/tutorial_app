Your First Component

Let's start with a simple example. This tutorial page itself is built with Element.js!

hello-world.js 
import { createComponent$ } from './element.js';

createComponent$('hello-world', (el, pageState) => {
    el.define$({
        template$: () => `
            <div class="greeting">
                <h2>Hello, Element.js!</h2>
                <p>Welcome to the future of web development</p>
            </div>
        `,
        style$: () => `
            [component] .greeting {
                padding: 20px;
                background: linear-gradient(135deg, #667eea 0%, #764ba2 100%);
                color: white;
                border-radius: 8px;
                text-align: center;
            }
        `,
        onInit$: async () => {
            console.log('Hello World component initialized!');
        }
    });
});
